访问者模式

第一种使用场景(通过call来借用别人的方法来给自己使用)

          let a = {
            name: 'zhanghui',
            addAttr() {
              this.age = 18;
              this.sex = "男"
            }
          }
          a.addAttr() // {name: "zhanghui", addAttr: ƒ, age: 18, sex: "男"}
          let b = { weight: '50kg' }
          a.addAttr.call(b) //{weight: "50kg", age: 18, sex: "男"}

第二种使用场景(让对象也有push方法)

        let vistor = {
          push:function(obj,value){
            Array.prototype.push.call(obj,value)
          }
        }
        let c ={}
        vistor.push(c,1000000)
